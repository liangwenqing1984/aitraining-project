// 🔧 直接更新数据库进度（多组合场景）
const progressPercent = Math.round((currentCombination / totalCombinationCount) * 99);
try {

    UPDATE tasks

### **TaskService的进度推送逻辑**

```typescript
if (isMultiCombination) {
  // 多组合：从数据库读取（由爬虫维护）
  const taskInfo = await db.prepare('SELECT progress FROM tasks WHERE id = $1').get(taskId);
  progressPercent = taskInfo?.progress || 0;  // ❌ 如果爬虫没更新，这里就是0
} else {
  // 单组合：基于recordCount估算
  if (totalRecords <= 10) {
    progressPercent = (totalRecords / 10) * 50;
  } else if (totalRecords <= 50) {
    progressPercent = 50 + ((totalRecords - 10) / 40) * 30;
  } else {
    progressPercent = 80 + Math.min(19, (totalRecords - 50) / 10);
  }
}

// 推送给前端
io.to(`task:${taskId}`).emit('task:progress', {
  taskId,
  status: 'running',
  progress: progressPercent,  // ❌ 这里是0
  current: totalRecords,      // ✅ 这里在增加
  recordCount: totalRecords,
  ...
});
```
